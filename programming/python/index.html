Import Data

Import Data from SQL Server into Pandas

import pyodbc
import pandas as pd
conn = pyodbc.connect('Driver={SQL Server};'
                      'Server=localhost;'
                      'Database=DataBase;'
                      'Trusted_Connection=yes;')

query = "select * from DataBase.dbo.annual_demand"
df = pd.read_sql(query, conn)

conn.close()

EDA

count null values

# count by columns
df.isnull().sum(axis=0)
# count by rows
df.isnull().sum(axis=1)
