This code helps me remove duplicate files, ending with -copy in the folder. import os arr = os.listdir() target = [] for i in arr: if "- Copy.JPG" in i or "- Copy.jpg" in i or "- Copy.heic" in i or "- Copy.BMP" in i or "- Copy.PNG" in i or "- Copy.png" in i: target.append(i) for t in target: os.remove(t)
